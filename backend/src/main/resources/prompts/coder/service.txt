```java
public Page<UserDTO> findAll(int page, int size) {

```

```java
// ✅ 正确 - 使用 MyBatis-Plus 分页
import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.baomidou.mybatisplus.core.metadata.IPage;

public IPage<UserDTO> findAll(int page, int size) {
    Page<UserEntity> entityPage = new Page<>(page, size);
    IPage<UserEntity> result = userMapper.selectPage(entityPage, null);
    Page<UserDTO> dtoPage = new Page<>(result.getCurrent(), result.getSize(), result.getTotal());
    dtoPage.setRecords(convertToDTO(result.getRecords()));
    return dtoPage;
}
```

```java
// ✅ 正确 - 使用 BeanUtils 手动转换
import org.springframework.beans.BeanUtils;

private UserDTO toDTO(UserEntity entity) {
    if (entity == null) return null;
    UserDTO dto = new UserDTO();
    BeanUtils.copyProperties(entity, dto);
    return dto;
}
```
